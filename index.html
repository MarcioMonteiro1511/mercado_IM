const express = require("express"); const sqlite3 = require("sqlite3").verbose(); const bcrypt = require("bcrypt"); const jwt = require("jsonwebtoken"); const app = express(); app.use(express.json()); // BANCO const db = new sqlite3.Database("mercado.db"); db.serialize(() => { db.run("CREATE TABLE IF NOT EXISTS admins (id INTEGER PRIMARY KEY, email TEXT, senha TEXT)"); db.run("CREATE TABLE IF NOT EXISTS produtos (id INTEGER PRIMARY KEY, nome TEXT, preco REAL, estoque INTEGER)"); db.run("CREATE TABLE IF NOT EXISTS vendas (id INTEGER PRIMARY KEY, total REAL)"); }); // ADMIN PADRÃO (async () => { const hash = await bcrypt.hash("123", 10); db.run("INSERT OR IGNORE INTO admins (id,email,senha) VALUES (1,'admin@email.com',?)",[hash]); })(); // LOGIN app.post("/login", (req,res)=>{ const {email,senha} = req.body; db.get("SELECT * FROM admins WHERE email=?", [email], async (err,user)=>{ if(!user) return res.json({erro:"Usuário não encontrado"}); const ok = await bcrypt.compare(senha,user.senha); if(!ok) return res.json({erro:"Senha inválida"}); const token = jwt.sign({id:user.id},"segredo"); res.json({token}); }); }); // AUTH function auth(req,res,next){ const token = req.headers.authorization; if(!token) return res.sendStatus(401); try{ jwt.verify(token,"segredo"); next(); } catch{ res.sendStatus(403); } } // PRODUTOS app.get("/produtos", auth, (req,res)=>{ db.all("SELECT * FROM produtos",[],(e,r)=>res.json(r)); }); app.post("/produtos", auth, (req,res)=>{ const {nome,preco,estoque} = req.body; db.run("INSERT INTO produtos (nome,preco,estoque) VALUES (?,?,?)", [nome,preco,estoque], ()=>res.send("OK")); }); // VENDA app.post("/venda", auth, (req,res)=>{ const {itens} = req.body; let total = 0; itens.forEach(i => total += i.subtotal); db.run("INSERT INTO vendas (total) VALUES (?)",[total]); itens.forEach(i=>{ db.run("UPDATE produtos SET estoque = estoque - ? WHERE id=?", [i.quantidade,i.id]); }); res.send("OK"); }); // DASHBOARD app.get("/dashboard", auth, (req,res)=>{ db.get("SELECT SUM(total) as total FROM vendas",(e,r)=>{ res.json({faturamento: r.total || 0}); }); }); // FRONTEND LIMPO E FUNCIONAL app.get("/", (req,res)=>{ res.send(` Mercado

Login

Dashboard

Faturamento: R$ 0

Produto

Caixa

    Total: R$ 0
    `); }); app.listen(3000,()=>console.log("Rodando em http://localhost:3000"));
