const express = require("express"); const sqlite3 = require("sqlite3").verbose(); const bcrypt = require("bcrypt"); const jwt = require("jsonwebtoken"); const app = express(); app.use(express.json({ limit: "10mb" })); const db = new sqlite3.Database("banco.db"); // 🔹 CRIAR TABELAS db.serialize(() => { db.run(`CREATE TABLE IF NOT EXISTS admins ( id INTEGER PRIMARY KEY AUTOINCREMENT, nome TEXT, email TEXT UNIQUE, senha TEXT )`); db.run(`CREATE TABLE IF NOT EXISTS clientes ( id INTEGER PRIMARY KEY AUTOINCREMENT, nome TEXT, cpf TEXT, foto TEXT, limite_mensal INTEGER DEFAULT 10 )`); db.run(`CREATE TABLE IF NOT EXISTS produtos ( id INTEGER PRIMARY KEY AUTOINCREMENT, nome TEXT, quantidade INTEGER, foto TEXT )`); db.run(`CREATE TABLE IF NOT EXISTS movimentacoes ( id INTEGER PRIMARY KEY AUTOINCREMENT, cliente_id INTEGER, produto_id INTEGER, tipo TEXT, quantidade INTEGER, data DATETIME DEFAULT CURRENT_TIMESTAMP )`); }); // 🔐 CRIAR ADMIN PADRÃO const criarAdmin = async () => { const senhaHash = await bcrypt.hash("123", 10); db.run(`INSERT OR IGNORE INTO admins (nome,email,senha) VALUES ('Admin','admin@email.com',?)`, [senhaHash]); }; criarAdmin(); // 🔐 LOGIN app.post("/login", (req, res) => { const { email, senha } = req.body; db.get("SELECT * FROM admins WHERE email=?", [email], async (err, user) => { if (!user) return res.json({ erro: "Usuário não encontrado" }); const ok = await bcrypt.compare(senha, user.senha); if (!ok) return res.json({ erro: "Senha inválida" }); const token = jwt.sign({ id: user.id }, "segredo"); res.json({ token }); }); }); // 🔐 MIDDLEWARE function auth(req, res, next) { const token = req.headers.authorization; if (!token) return res.sendStatus(401); try { jwt.verify(token, "segredo"); next(); } catch { res.sendStatus(403); } } // 👤 CLIENTES app.post("/clientes", auth, (req, res) => { const { nome, cpf, foto } = req.body; db.run("INSERT INTO clientes (nome,cpf,foto) VALUES (?,?,?)", [nome, cpf, foto], () => res.send("OK")); }); app.get("/clientes", auth, (req, res) => { db.all("SELECT * FROM clientes", [], (err, rows) => res.json(rows)); }); // 📦 PRODUTOS app.post("/produtos", auth, (req, res) => { const { nome, quantidade, foto } = req.body; db.run("INSERT INTO produtos (nome,quantidade,foto) VALUES (?,?,?)", [nome, quantidade, foto], () => res.send("OK")); }); app.get("/produtos", auth, (req, res) => { db.all("SELECT * FROM produtos", [], (err, rows) => res.json(rows)); }); // 🔄 ENTRADA app.post("/entrada", auth, (req, res) => { const { produto_id, quantidade } = req.body; db.run("UPDATE produtos SET quantidade = quantidade + ? WHERE id=?", [quantidade, produto_id]); db.run("INSERT INTO movimentacoes (produto_id,tipo,quantidade) VALUES (?,?,?)", [produto_id, "entrada", quantidade]); res.send("OK"); }); // 🔻 SAÍDA (com controle mensal) app.post("/saida", auth, (req, res) => { const { cliente_id, produto_id, quantidade } = req.body; db.get(` SELECT SUM(quantidade) as total FROM movimentacoes WHERE cliente_id=? AND tipo='saida' AND strftime('%m', data)=strftime('%m','now') `, [cliente_id], (err, result) => { const total = result.total || 0; db.get("SELECT limite_mensal FROM clientes WHERE id=?", [cliente_id], (e, cli) => { if (total + quantidade > cli.limite_mensal) { return res.json({ erro: "Limite mensal excedido" }); } db.run("UPDATE produtos SET quantidade = quantidade - ? WHERE id=?", [quantidade, produto_id]); db.run("INSERT INTO movimentacoes (cliente_id,produto_id,tipo,quantidade) VALUES (?,?,?,?)", [cliente_id, produto_id, "saida", quantidade]); res.send("OK"); }); }); }); // 📊 RELATÓRIO app.get("/relatorio/:id", auth, (req, res) => { db.all("SELECT * FROM movimentacoes WHERE cliente_id=?", [req.params.id], (err, rows) => res.json(rows)); }); // 🌐 FRONTEND BONITO app.get("/", (req, res) => { res.send(` Supermercado Solidário

Login

Clientes

Produtos

Movimentação

Comprovante

`); }); app.listen(3000, () => console.log("Rodando em http://localhost:3000"));
